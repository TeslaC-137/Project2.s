.data
	theString: .space 2500

	#Error Messages
	INVALID_CHAR: .asciiz "Invalid base-27 number."
	EMPTY: .asciiz "Input is empty."
	TOO_LONG: .asciiz "Invalid base-N number."

.text
	main:
		#Taking user input as string
		li $v0, 8
		la $a0, theString
		li $a1, 2500
		syscall

		la $t0, theString
		lb $t1, 0($t0)
		li $t2, 10
		beq $t1, $t2, empty

	exit:
		#Exiting the program
		li $v0, 10
		syscall
